\subsection{定义}

\end{equation}

h(x)exp(\theta^T\phi(x)-A(\theta))
\end{equation}

\subsection{对数配分函数}

\begin{equation}
\frac{dA}{d\theta} = E[\phi(x)]
\end{equation}

\begin{equation}
\frac{d^2A}{d\theta^2} = var[\phi(x)]
\end{equation}

\begin{equation}
\frac{\partial^2A}{\partial\theta_i\partial\theta_j}=
E[\phi_i(x)\phi_j(x)] - E[\phi_i(x)]E[\phi_j(x)]
\end{equation}

\begin{equation}
\nabla A(\theta) = cov[\phi(x)]
\end{equation}
